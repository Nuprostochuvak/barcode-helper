function doGet() {
  var sheet = SpreadsheetApp.openById("16eiDJ85tUdku-IdwhEigfBGpRuhz-xgn6KXWJQJaJ5I");

  var values = sheet.getActiveSheet().getDataRange().getValues();
  var data = [];

  for (var i=1; i< values.length; i++){
    var row = values[i];

    var barcode = {};


    barcode['id'] = row[0]; //name
    barcode['name'] = row[1]; //source
    barcode['codeOfBarcode'] = row[2]; // barcode
    barcode['description'] = row[3]; // sourceUrl
    data.push(barcode);

  }

  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}
